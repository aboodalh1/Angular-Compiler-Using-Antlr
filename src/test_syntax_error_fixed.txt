// Correct Syntax Test File - Demonstrates proper ANTLR grammar usage
import { Component } from '@angular/core';

@Component()
export class TestComponent {
    // Variable declarations with proper syntax
    title: string = "Test Component";
    showMessage: boolean = true;
    count: number = 42;
    
    // Array declaration without generics (as per grammar)
    items: Array = ["Item 1", "Item 2", "Item 3"];
    
    // Object declaration
    config: any = { enabled: true, timeout: 5000 };
    
    // Function declaration
    onClick() {
        console.log("Button clicked!");
        this.count = this.count + 1;
    }
    
    // Method with parameters
    updateTitle(newTitle: string) {
        this.title = newTitle;
    }
    
    // Conditional statement
    isVisible(): boolean {
        if (this.showMessage && this.count > 0) {
            return true;
        } else {
            return false;
        }
    }
    
    // Loop statement
    processItems() {
        let i = 0;
        while (i < this.items.length) {
            console.log(this.items[i]);
            i = i + 1;
        }
    }
}
