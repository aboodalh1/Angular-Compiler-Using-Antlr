@Component({
  selector: 'app-product-manager'
})
export class ProductManagerComponent {
  products = [];
  currentProduct = null;
  
  addProduct(product) {
    this.products.push(product);
  }
  
  deleteProduct(id) {
    this.products = this.products.filter(p => p.id !== id);
  }
  
  selectProduct(id) {
    this.currentProduct = this.products.find(p => p.id === id) || null;
  }
}
